Cosmos:=LoadImage('/Cosmos.png');

X:=50;

while (Key<>GA_FIRE) do
begin
DrawImage(Cosmos,0,0); DrawImage(NLO,X,Y);
Delay(100);
repaint;
Key:=KeyToAction(getKeyPressed); if (Key=GA_LEFT) then X:=X-3;
if (Key=GA_RIGHT) then X:=X+3;
if (Key=GA_UP) then Y:=Y-3;
if (Key=GA_DOWN) then Y:=Y+3;
end;
end.
